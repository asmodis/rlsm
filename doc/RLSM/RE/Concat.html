def follow
  result = []

  (@content.size-1).times do |i|
    result |= @content[i].follow
    @content[i].last.each do |char1|
      @content[i+1].first.each do |char2|
        result << [char1, char2]
      end
    end
  end

  result |= @content[-1].follow

  result.sort
end

def to_s
  string = ''
  
  @content.each do |subexpr|
    if Union === subexpr
      string += "(#{subexpr.to_s})"
    else
      string += subexpr.to_s
    end
  end

  string
end
